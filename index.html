graphcalc-api

A RESTful API for graph-calc.

Installation: npm install graphcalc-api

Usage: 

var graphCalcApiServer = require('graphcalc-api');
var MyGraph = require('./graph'); // your graphcalc implementation

var graph = graphCalcApiServer({
  port: 8080,
  graph: new MyGraph()
});
